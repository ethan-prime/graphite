[ EXPRESSIONS ]
<stmt> ::= <function_definition> | <function_call> | <variable_declaration> | <assignment>
<expr> ::= <primary_expr> | <binary_expr> [<operator> <expr>]*
<primary_expr> ::= <double> | <function_call> | <identifier>
<binary_expr> ::= <primary_expr> <operator> <primary_expr>
<operator> ::= + | - | * | /

[ FUNCTIONS ]
<function_definition> ::= def <identifier> (<parameters>) => <type> { [<expr>]* }
<funcion_call> ::= <identifier> (<expr> [, <expr>]*)
<type> ::= dbl
<parameters> ::= <argument> [, <argument]*
<parameter> ::= <type> <identifier>

[ VARIABLES ]
<variable_declaration> ::= let <identifier> = <expr>
<assignment> ::= <identifier> = <expr>
